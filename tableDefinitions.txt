CREATE TABLE Borrower(
      bid INT,
      password INT,
      name NVARCHAR(20),
      address NVARCHAR(40),
      phone INT,
      emailAddress NVARCHAR(40),
      sinOrStNo INT,
      expiryDate INT,
      type NVARCHAR(20),
      PRIMARY KEY(bid)
)
GO
CREATE TABLE BorrowerType (
      type NVARCHAR(20),
      bookTimeLimit INT,
      PRIMARY KEY(type)
)
GO
